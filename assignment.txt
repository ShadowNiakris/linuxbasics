#1. Прочитайте в man-странице команды sort про возможности параметра -k, --key и напишите команду, сортирующую 
#файл, представленный ниже по третьей цифре, в третьей колонке. В качестве разделителя используйте двоеточие. 
#Для наглядности используйте параметр --debug.

#root:x:100:
#bin:x:131:
#daemon:x:122:

sort -nt ":" --key=3.3 --debug file1.txt

#2. Напишите конвейер команд, который выводит распределение стандартных оболочек пользователей в порядке убывания. 
#Подсказка: используйте файл /etc/passwd, cut, sort и uniq.
#Ожидаемый вывод команды (кол-во оболочек может отличаться)
#
#29 /usr/sbin/nologin
#...
#1 /bin/sync

cat /etc/passwd | cut -d ":" -f 7 | sort | uniq -c | sort -nr

#3. Напишите команду, которая считает количество коммитов в тексте по ссылке 
#https://cdn.kernel.org/pub/linux/kernel/v4.x/ChangeLog-4.4.44.
#Подсказка: Можно считать строки начинающиеся с commit

wget https://cdn.kernel.org/pub/linux/kernel/v4.x/ChangeLog-4.4.44 -q -O -
| grep ^commit | wc -l

#4. Используя ссылку из предыдущей задачи, посчитайте распределение коммитов по месяцам.
#Вывод должен быть суммарный по каждому месяцу.
wget https://cdn.kernel.org/pub/linux/kernel/v4.x/ChangeLog-4.4.44 -q -O - | grep ^Date: | cut -d ' ' -f 5 | sort | uniq -c | sort -nr
