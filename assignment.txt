#Задание:
#1.Используя awk, распечатайте из /etc/passwd имя текущего пользователя и его командную оболочку. 
#Постарайтесь динамически определять имя пользователя, используя команды или другие способы, доступные в оболочке.
```
awk -v v="$(whoami)" -F ':' '{ if ($1 == v) print $1 " " $7}' /etc/passwd
```

#2.Используя awk, посчитайте в /etc/passwd сумму всех идентификаторов пользователей, 
#которые используют командную оболочку bash.
```
awk -F ':' '{if ($7 == "/usr/sbin/nologin") n=n+$3 } END {print n}' /etc/passwd
```

#3. Используя awk, распечатайте четные строки файла /etc/passwd.
```
awk '{if (NR % 2 == 0) print  }' /etc/passwd
```

#4. Используя функцию length языка awk, которая считает длину переданного ей аргумента, 
#распечатайте все строки файла /etc/passwd, у которых имя пользователя длиннее четырех символов.
```
awk -F ':' '{if (length($1)>4) print}' /etc/passwd
```
